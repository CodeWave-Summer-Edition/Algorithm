class Solution {
    public int solution(int a, int b, int current) {
        int answer = 0;
        
        int get = 0;
        int rest = 0;
        
        while(current >= a){
            get = (current/a)*b;
            rest = current%a;
            
            answer += get;
            current = get+rest;
        }
        
        return answer;
    }
}
